# PayloadCMS Environment Configuration Template
# Copy this file to .env for local development

# ============================================
# Database Configuration
# ============================================

# Shared Supabase PostgreSQL Database with separate schema
# PayloadCMS uses 'payload' schema, spoke-hire-web uses 'public' schema
# This provides isolation while sharing the same database
# Get connection string from: https://app.supabase.com/project/[PROJECT-REF]/settings/database
# Use "Transaction" mode (pgbouncer=true) for DATABASE_URL
DATABASE_URL="postgresql://postgres.[PROJECT-REF]:[PASSWORD]@aws-0-[REGION].pooler.supabase.com:5432/postgres?pgbouncer=true"

# Direct Connection URL (for migrations - Supabase only)
# Use "Direct connection" for DIRECT_URL
DIRECT_URL="postgresql://postgres.[PROJECT-REF]:[PASSWORD]@aws-0-[REGION].pooler.supabase.com:5432/postgres"

# Legacy DATABASE_URI (for compatibility with PayloadCMS defaults)
# Can use same value as DATABASE_URL
DATABASE_URI="${DATABASE_URL}"

# Database Schema Name for PayloadCMS
# PayloadCMS will create all tables in this schema
# Default: "payload" (if not specified)
# This provides logical isolation from spoke-hire-web tables (which use 'public' schema)
PAYLOAD_DB_SCHEMA="payload"

# ============================================
# PayloadCMS Configuration
# ============================================

# Secret key for PayloadCMS (generate a random string)
# Used for encrypting sessions and tokens
# Generate with: openssl rand -base64 32
PAYLOAD_SECRET="your-payload-secret-key-here"

# Server URL where PayloadCMS will be accessible
# Development: "http://localhost:3000"
# Production: "https://cms.your-domain.com"
PAYLOAD_PUBLIC_SERVER_URL="http://localhost:3000"

# ============================================
# Supabase Storage S3 Configuration (PRODUCTION ONLY)
# ============================================
# NOTE: These are only required for production deployment
# In local development (localhost), PayloadCMS uses local filesystem storage automatically
# Files are stored in: ./media/ directory

# Supabase Storage S3 Endpoint
# Format: https://[PROJECT-REF].supabase.co/storage/v1/s3
# Get from: https://app.supabase.com/project/[PROJECT-REF]/settings/storage
# Only needed in production (NODE_ENV=production)
SUPABASE_STORAGE_ENDPOINT="https://[PROJECT-REF].supabase.co/storage/v1/s3"

# Supabase Storage Region
# Typically "us-east-1" or your Supabase project region
# Only needed in production
SUPABASE_STORAGE_REGION="us-east-1"

# S3 Access Key ID from Supabase Storage
# Get from: Supabase Dashboard > Storage > Settings > S3 API
# Only needed in production
SUPABASE_ACCESS_KEY_ID="your-supabase-access-key-id"

# S3 Secret Access Key from Supabase Storage
# Get from: Supabase Dashboard > Storage > Settings > S3 API
# Only needed in production
SUPABASE_SECRET_ACCESS_KEY="your-supabase-secret-access-key"

# Storage Bucket Name
# Create a bucket in Supabase Storage and use its name here
# Only needed in production
SUPABASE_BUCKET_NAME="payload-media"

# ============================================
# Environment
# ============================================

NODE_ENV="development"

# ============================================
# Notes
# ============================================

# Setup Instructions:
# 1. Copy this file: cp env.example.txt .env
# 2. Update DATABASE_URL with your Supabase PostgreSQL connection (same as spoke-hire-web)
# 3. Generate PAYLOAD_SECRET (use: openssl rand -base64 32)
# 4. For LOCAL DEVELOPMENT: No storage configuration needed - uses local filesystem (./media/)
# 5. For PRODUCTION: Configure Supabase Storage S3 credentials (see below)
# 6. Update PAYLOAD_PUBLIC_SERVER_URL for your deployment URL
# 7. The 'payload' schema will be created automatically on first migration

# Getting Supabase Storage S3 Credentials:
# 1. Go to: https://app.supabase.com/project/[PROJECT-REF]/settings/storage
# 2. Enable S3 API if not already enabled
# 3. Generate S3 credentials (Access Key ID and Secret Access Key)
# 4. Create a storage bucket for PayloadCMS media files
# 5. Use the bucket name in SUPABASE_BUCKET_NAME

# Database Notes:
# - PayloadCMS uses a configurable schema (default: 'payload') in the same PostgreSQL database as spoke-hire-web
# - Set PAYLOAD_DB_SCHEMA to customize the schema name (defaults to 'payload' if not set)
# - This provides logical isolation while sharing the same database instance
# - spoke-hire-web tables are in 'public' schema, PayloadCMS tables in the configured schema
# - PostgreSQL schemas are a standard way to organize database objects
# - The schema will be created automatically when you run migrations
# - DATABASE_URI is included for compatibility with PayloadCMS defaults

# Storage Notes:
# - LOCAL DEVELOPMENT (NODE_ENV=development):
#   * Uses local filesystem storage automatically
#   * Files stored in: ./media/ directory
#   * No S3 configuration needed
#   * Perfect for local testing and development
# - PRODUCTION (NODE_ENV=production):
#   * Uses Supabase Storage via S3-compatible API
#   * Requires all SUPABASE_STORAGE_* environment variables
#   * Files stored in Supabase Storage bucket
#   * Configured automatically based on NODE_ENV
